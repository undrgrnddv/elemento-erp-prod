

----------------------------------------------------------------------------------------------------
INSTALL GIT
----------------------------------------------------------------------------------------------------
sudo apt update
sudo apt install git

----------------------------------------------------------------------------------------------------
GENERATE SSH KEY
----------------------------------------------------------------------------------------------------
Step 1: Generate an SSH Key Pair
ssh-keygen -t ed25519 -C "c@elementorural.com"
Step 2: Copy the Public Key
cat ~/.ssh/id_ed25519.pub
Step 3: Add the Public Key to GitHub
    Click the New SSH key button.
    Title: Give it a descriptive name (e.g., "Instance-2025 Server").
    Key: Paste the entire key you copied in Step 2 into the large text box.

----------------------------------------------------------------------------------------------------
CLONE REPO
----------------------------------------------------------------------------------------------------
Run the Clone Command
git clone git@github.com:undrgrnddv/00_Elemento_ERP_Local.git

----------------------------------------------------------------------------------------------------
INSTALL Docker
----------------------------------------------------------------------------------------------------
Command to Install Docker
sudo apt update && sudo apt install docker.io
sudo apt install docker-compose

----------------------------------------------------------------------------------------------------
SET PERMISSIONS
----------------------------------------------------------------------------------------------------
sudo usermod -aG docker $USER
newgrp docker

----------------------------------------------------------------------------------------------------
START CONTAINERS
----------------------------------------------------------------------------------------------------
docker-compose up -d
